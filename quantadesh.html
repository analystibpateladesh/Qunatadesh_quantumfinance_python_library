from quantadesh.models.black_scholes import bs_price, bs_implied_vol
from quantadesh.models.binomial       import binomial_price
from quantadesh.models.monte_carlo    import mc_price
from quantadesh.models.heston         import heston_price
from quantadesh.models.bachelier      import bachelier_price
from quantadesh.models.black76        import black76_price

S, K, T, r, sigma = 100, 105, 1.0, 0.05, 0.20

# Black-Scholes - handles call and put via option_type
call = bs_price(S, K, T, r, sigma, option_type='call')
put  = bs_price(S, K, T, r, sigma, option_type='put')

# Binomial Tree (200 steps)
bc = binomial_price(S, K, T, r, sigma, steps=200, option_type='call')

# Monte Carlo
mc = mc_price(S, K, T, r, sigma, option_type='call', paths=100_000, seed=42)

# Heston Stochastic Volatility
hc = heston_price(S, K, T, r, v0=0.04, kappa=2.0,
                   theta=0.04, sigma=0.3, rho=-0.7, option_type='call')

# Bachelier - sigma is absolute (normal) vol
bac = bachelier_price(S, K, T, r, sigma=S*sigma, option_type='call')

# Black-76 - for futures/bond options; F is the forward price
F   = S * (1 + r)**T
b76 = black76_price(F, K, T, r, sigma, option_type='call')

# Implied Volatility solver
iv  = bs_implied_vol(8.50, S, K, T, r, option_type='call')

# Asian arithmetic average

                           average_type='arithmetic', paths=50_000, seed=42)

# Asian geometric average
asian_geo = asian_price(S, K, T, r, sigma, option_type='call',
                          average_type='geometric', paths=50_000, seed=42)

# Barrier: Up-and-Out (knocked out if spot >= H=120)
uao = barrier_price(S, K, H=120, T=T, r=r, sigma=sigma,
                     barrier_type='up-and-out', option_type='call', seed=42)

# Barrier: Down-and-In (activated if spot <= H=85)
dai = barrier_price(S, K, H=85, T=T, r=r, sigma=sigma,
                     barrier_type='down-and-in', option_type='put', seed=42)

# Lookback: floating strike, call pays S_T - min(path)
lb = lookback_price(S, T, r, sigma, option_type='call', paths=50_000, seed=42)

# Digital: pays $1 cash if S_T > K
dig = digital_price(S, K, T, r, sigma, cash=1.0, option_type='call')

r0 = 0.05

# Vasicek: dr = a(b-r)dt + σ dW
v_paths = vasicek_simulate(r0=r0, a=0.5, b=0.05, sigma=0.02,
                           T=5, steps=252*5, paths=1000, seed=42)
zcb_v = vasicek_zcb_price(r0=r0, a=0.5, b=0.05, sigma=0.02, T=5)

# CIR: dr = a(b-r)dt + σ√r dW - no negative rates
c_paths = cir_simulate(r0=r0, a=0.5, b=0.05, sigma=0.1,
                        T=5, steps=252*5, paths=1000, seed=42)
zcb_c = cir_zcb_price(r0=r0, a=0.5, b=0.05, sigma=0.1, T=5)

# Hull-White: dr = (θ(t) - a·r)dt + σ dW
# θ can be a float (constant) or a callable θ(t)
hw_paths = hull_white_simulate(r0=r0, a=0.5, sigma=0.02,
                               theta=0.03, T=5, steps=252*5, paths=1000, seed=42)

04

Visualisation

quantadesh.plot.charts
Chart 01
Option Payoff
Payoff diagram at expiry with breakeven line. Shows gross payoff and P&L net of premium. Supports call and put, any strike and premium.
plot_payoff(K, option_type, premium)
Chart 02
Monte Carlo GBM Paths
Simulates and plots n geometric Brownian motion paths. Reference line marks the initial price S₀. Useful for visualising uncertainty over a horizon.
plot_mc_paths(S, T, r, sigma, paths, seed)
Chart 03
Implied Volatility Smile
Inverts market prices across a strike range to plot the IV smile. Pass a dict of {strike: market_price}. Reveals skew, smirk, and term structure.
plot_vol_smile(S, T, r, prices, option_type)
Chart 04
Efficient Frontier
Scatter plot of the mean-variance frontier with points coloured by Sharpe ratio. Maximum Sharpe portfolio marked with a star. Pass a FrontierResult object.
plot_efficient_frontier(frontier, rf)
Chart 05
Drawdown
Two-panel chart: cumulative wealth curve above, drawdown series below with underwater periods shaded. Pass a daily returns array.
plot_drawdown(returns)
Chart 06
Yield Curve
Overlays zero rate curve (solid) and 3-month forward rate curve (dashed) with knot points marked. Pass a YieldCurve object and the max tenor to display.
plot_yield_curve(curve, t_max)
Chart 07
Greeks Surface
Heatmap of any BS Greek across a spot × time grid. Accepts greek='delta', 'gamma', 'vega', 'theta', or 'rho'. The delta surface is the most common use case.
plot_greeks_surface(K, T, r, sigma, option_type, greek)
Chart 08
P&L Distribution
Histogram of return distribution with VaR and CVaR vertical lines overlaid. Clearly shows the left tail and the gap between the two risk measures.
plot_pnl_distribution(returns, var_value, cvar_value)
